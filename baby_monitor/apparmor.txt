#include <tunables/global>

profile baby_monitor flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>
  #include <abstractions/ssl_certs>

  network inet stream,
  network inet6 stream,
  network unix stream,

  signal (receive),
  signal (send) set=(kill,term,int,hup,cont),

  /bin/sh rix,
  /usr/bin/ffmpeg Cx -> ffmpeg,
  /usr/local/bin/baby-monitor-run rix,
  /usr/local/bin/python* rix,

  /app/** r,
  /data/ rw,
  /data/** rwk,
  /tmp/ rw,
  /tmp/** rwk,

  /dev/null rw,
  /dev/urandom r,
  /etc/localtime r,
  /etc/passwd r,
  /etc/group r,
  /proc/** r,
  deny /proc/{,**/}environ r,
  /sys/devices/system/cpu/** r,
  /usr/lib/** mr,
  /usr/local/lib/** mr,
  /usr/share/zoneinfo/** r,

  profile ffmpeg flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    #include <abstractions/nameservice>
    #include <abstractions/ssl_certs>

    network inet stream,
    network inet6 stream,
    network inet dgram,
    network inet6 dgram,
    network unix stream,

    signal (receive),
    signal (send) set=(kill,term,int,hup,cont),

    /usr/bin/ffmpeg mr,
    /usr/lib/** mr,
    /usr/share/ffmpeg/** r,
    /tmp/ rw,
    /tmp/** rwk,

    /dev/null rw,
    /dev/urandom r,
    /etc/localtime r,
    /proc/cpuinfo r,
    /proc/self/status r,
    deny /proc/{,**/}environ r,
    /sys/devices/system/cpu/** r,

    deny /app/** rwklx,
    deny /data/** rwklx,
  }
}
